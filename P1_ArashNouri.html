import math

def grayscale(img):
    """Applies the Grayscale transform
    This will return an image with only one color channel
    but NOTE: to see the returned image as grayscale
    (assuming your grayscaled image is called 'gray')
    you should call plt.imshow(gray, cmap='gray')"""
    return cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    # Or use BGR2GRAY if you read an image with cv2.imread()
    # return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
def canny(img, low_threshold, high_threshold):
    """Applies the Canny transform"""
    return cv2.Canny(img, low_threshold, high_threshold)

def gaussian_blur(img, kernel_size):
    """Applies a Gaussian Noise kernel"""
    return cv2.GaussianBlur(img, (kernel_size, kernel_size), 0)

def region_of_interest(img, vertices):
    """
    Applies an image mask.
    
    Only keeps the region of the image defined by the polygon
    formed from `vertices`. The rest of the image is set to black.
    `vertices` should be a numpy array of integer points.
    """
    #defining a blank mask to start with
    mask = np.zeros_like(img)   
    
    #defining a 3 channel or 1 channel color to fill the mask with depending on the input image
    if len(img.shape) > 2:
        channel_count = img.shape[2]  # i.e. 3 or 4 depending on your image
        ignore_mask_color = (255,) * channel_count
    else:
        ignore_mask_color = 255
        
    #filling pixels inside the polygon defined by "vertices" with the fill color    
    cv2.fillPoly(mask, vertices, ignore_mask_color)
    
    #returning the image only where mask pixels are nonzero
    masked_image = cv2.bitwise_and(img, mask)
    return masked_image


def draw_lines(img, lines, color=[255, 0, 0], thickness=2):
    """
    NOTE: this is the function you might want to use as a starting point once you want to 
    average/extrapolate the line segments you detect to map out the full
    extent of the lane (going from the result shown in raw-lines-example.mp4
    to that shown in P1_example.mp4).  
    
    Think about things like separating line segments by their 
    slope ((y2-y1)/(x2-x1)) to decide which segments are part of the left
    line vs. the right line.  Then, you can average the position of each of 
    the lines and extrapolate to the top and bottom of the lane.
    
    This function draws `lines` with `color` and `thickness`.    
    Lines are drawn on the image inplace (mutates the image).
    If you want to make the lines semi-transparent, think about combining
    this function with the weighted_img() function below
    """
    
    for line in lines:
        for x1, y1, x2, y2 in line:
            cv2.line( img, (x1, y1), (x2, y2), color, thickness )
            
def hough_lines(img, rho, theta, threshold, min_line_len, max_line_gap):
    """
    `img` should be the output of a Canny transform.
        
    Returns an image with hough lines drawn.
    """
    lines = cv2.HoughLinesP(img, rho, theta, threshold, np.array([]), minLineLength=min_line_len, maxLineGap=max_line_gap)
    line_img = np.zeros((img.shape[0], img.shape[1], 3), dtype=np.uint8)
    draw_lines(line_img, lines)
    return line_img

# Python 3 has support for cool math symbols.

def weighted_img(img, initial_img, =0.8, =1., =0.):
    """
    `img` is the output of the hough_lines(), An image with lines drawn on it.
    Should be a blank image (all black) with lines drawn on it.
    
    `initial_img` should be the image before any processing.
    
    The result image is computed as follows:
    
    initial_img *  + img *  + 
    NOTE: initial_img and img must be the same shape!
    """
    return cv2.addWeighted(initial_img, , img, , )

def Lane_Finder(img):
    '''
    Finds the lines in the interested area of an image by using Canny edge detection and Hough transform 
    '''
    
    # Apply grayscale transform in the image
    gray = grayscale( img )         
        
    # Apply Guassian noise kernel
    kernel_size = 3
    gray = gaussian_blur( gray, kernel_size )
        
    # Apply Canny transform
    low_threshold = 50
    high_threshold = 150
    edges = canny( gray, low_threshold, high_threshold )  
        
    # Masking
    '''
    imshape = img.shape                                                    # Shape of the image
    vertices = np.array( [ [ ( .51 * imshape[ 1 ], imshape[ 0 ] * .58 ),\
                            ( .49 * imshape[ 1 ], imshape[ 0 ] * .58 ),\
                            ( 0, imshape[ 0 ] ),\
                            ( imshape[ 1 ], imshape[ 0 ] ) ] ],\
                        dtype = np.int32 )
    '''
    vertices = np.array( [  [( 120, 545 ),\
                          ( 910, 545 ),\
                          ( 540, 325 ),\
                          ( 445, 325 ) ] ] )
    target = region_of_interest( edges, vertices )                      # Masking the image
    plot_region_of_interest( img, vertices )
    
    # Hough transfrom
    rho = 3                                                             # Distance resolution of the accumulator in pixels
    theta = np.pi / 180                                                 # Angle resolution of the accumulator in radians
    threshold = 48#19#23#15#30#                                         # Minimum numbe rof votes
    min_line_len = 7#40#60 #20#100#                                     # For the probabilistic Hough transform, it is the minimum line length in pixels 
    max_line_gap = 4#44#83 #5#160#                                      # For the probabilistic Hough transform, it is the maximum gap between line segments lying on the same line to treat them as a single line segment in pixels
    
    lines = hough_lines( target, rho, theta, threshold, min_line_len, max_line_gap )
    
    # Blending the results with the original image
    result = weighted_img( lines, img,  = .8,  = 1.0 )
    
    return result

def BGR2RGB( BGR ):
    '''
    Converts the CV2 outputs from BGR to RGB
    '''

    # Split input colors
    r, g, b = cv2.split( BGR )
    
    # Converting to rgb
    converted = cv2.merge( ( b, g, r ) )

    return converted

def draw_lines(img, lines, color=[255, 0, 0], thickness=2):
    """
    NOTE: this is the function you might want to use as a starting point once you want to 
    average/extrapolate the line segments you detect to map out the full
    extent of the lane (going from the result shown in raw-lines-example.mp4
    to that shown in P1_example.mp4).  
    
    Think about things like separating line segments by their 
    slope ((y2-y1)/(x2-x1)) to decide which segments are part of the left
    line vs. the right line.  Then, you can average the position of each of 
    the lines and extrapolate to the top and bottom of the lane.
    
    This function draws `lines` with `color` and `thickness`.    
    Lines are drawn on the image inplace (mutates the image).
    If you want to make the lines semi-transparent, think about combining
    this function with the weighted_img() function below
    """
    imshape = img.shape
    rm = []                                                                 # Right slopes
    lm = []                                                                 # Left slopes
    rc = []                                                                 # Right centers
    lc = []                                                                 # left centers
    
    xl = []                                                                 # Left xs
    xr = []                                                                 # Right xs
    yl = []                                                                 # Left ys
    yr = []                                                                 # Right ys
    
    for line in lines:
        for x1, y1, x2, y2 in line:
            cv2.line( img, (x1, y1), (x2, y2), color, thickness )
            slope = ( y2 - y1 ) / ( x2 - x1 )
            centers = [ ( x2 + x1 ) / 2, ( y2 + y1 ) / 2 ]
            
            # Seperate the right from left side points.
            # Small negative values show the left side points
            if slope < -.5:
                lm.append( slope )
                lc.append( centers )
                xl.append( x2 )
                xl.append( x1 )
                yl.append( y2 )
                yl.append( y1 )
                
            # Small posetive values show the right side points    
            elif slope > .5:
                rm.append( slope )
                rc.append( centers )
                xr.append( x2 )
                xr.append( x1 )
                yr.append( y2 )
                yr.append( y1 )
    
    # Finding the average of slopes for both right and left
    r_slope = np.sum( rm ) / len( rm )
    l_slope = np.sum( lm ) / len( lm )
    
    # Finding the average coordinate of centers for both right and left
    r_center = np.divide( np.sum( rc, axis = 0 ), len( rc ) )
    l_center = np.divide( np.sum( lc, axis = 0 ), len( lc ) )
    
    # Defining Max height for the line
    ylmax = int(max(yl))                                                    # Max height for the left side
    yrmax = int(max(yr))                                                    # Max height for the right side
    ymax = max(ylmax,yrmax)                                                 # Max height between left nd right side. This value going to be used for both sides
    
    xlmax = int(((-l_center[1]+ymax) + (l_slope * l_center[0]))/l_slope)    # Finding the Max x on the left
    xrmax = int(((-r_center[1]+ymax) + (r_slope * r_center[0]))/r_slope)    # Finding the Max x on the right
    
    ylmin = int(min(yl))                                                    # min height for the left side
    yrmin = int(min(yr))                                                    # min height for the right side
    ymin = min(ylmin,yrmin)                                                 # min height between left nd right side. This value going to be used for both sides
    
    xlmin = int(((-l_center[1]+ymin) + (l_slope * l_center[0]))/l_slope)    # Finding the min x on the left 
    xrmin = int(((-r_center[1]+ymin) + (r_slope * r_center[0]))/r_slope)    # Finding the min x on the right 
    
    
    
    cv2.line(img, (xlmax, ymax),(xlmin, ymin),(255,0,0),10)                 # Drawing the line on the left side
    cv2.line(img, (xrmax, ymax),(xrmin, ymin),(255,0,0),10)                 # Drawing the line on the right side
